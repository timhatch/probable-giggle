NON-JSON Responses

Think about eliminating the try:catch in the m.request implementations by using the "extract" option
see http://mithril.js.org/mithril.request.html for clues on implementation

var nonJSONError = function(xhr) {
  return xhr.status > 200 ? JSON.stringify(xhr.responseText) : xhr.responseText
}

m.request({
  method: "GET", 
  url: "/foo/bar.x", 
  extract: nonJSONError
})
.then(function(data) {}, function(error) {console.log(error)})
